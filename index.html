會員登入: bat.identify

名稱 值類型 是否必須提供 說明 id 為欲追蹤之會員帳號，以vip-user-1234 為例
id string 是  id 為會員登入後其代表的一組帳號或是字串 
bat.identify('id');
name string 否  name 為會員登入後額外形容其輪廓的名稱 
bat.identify('vip-user-1234',{name: 'BAT'})
email string 否  email 為會員登入後額外形容其輪廓的電子信箱 
bat.identify('vip-user-1234',{email: 'email@email.com'})
sex string 否  sex 為會員登入後額外形容其輪廓的性別 
bat.identify('vip-user-1234',{sex: 'male or female'})
age number 否  age 為會員登入後額外形容其輪廓的年齡 
bat.identify('vip-user-1234',{age: 20})
tel number 否  tel 為會員登入後額外形容其輪廓的室內電話 
bat.identify('vip-user-1234',{tel: 0225667789})
mobile number 否  mobile 為會員登入後額外形容其輪廓的手機號碼 
bat.identify('vip-user-1234',{mobile: 0911222333})
address string 否  tel 為會員登入後額外形容其輪廓的地址 
bat.identify('vip-user-1234',{address: '台北市松山區民權東路三段144號'})

使用範例

bat.identify('vip-user-1234',{email: 'abc@example.com', age:30, sex: 'male'});

單一商品頁面: SingleItemNo

名稱 值類型 是否必須提供 說明 使用範例: variable 為追蹤之變數值
SingleItemNo number 是 商品編號 
bat.track('SingleItemNo',{ SingleItemNo: variable});
name string 否 商品名稱 
bat.track('SingleItemNo',{ name: variable});
price number 否 商品價格 
bat.track('SingleItemNo',{ price: variable});
category[i]
i為從1開始的正整數		 string 否 商品分類
可能有大分類,中分類,小分類		 
bat.track('SingleItemNo',{ category1: variable,category2: variable});
label[i]
i為從1開始的正整數		 string 否 商品標籤
可能有顏色或是size		 
bat.track('SingleItemNo',{ label1: variable,label2: variable});
content string 否 商品內容 
bat.track('SingleItemNo',{ content: variable});

使用範例

bat.track('SingleItemNo',{ SingleItemNo:'12345678',category1:'大衣',label1:'黑色',name: '超值防寒大衣', content: '超值選擇,最後一天'});

自定義頁面:CustomValue

名稱 值類型 是否必須提供 說明 使用範例: variable 為變數值
title string 是 該頁面的自定義標題值 
bat.track('CustomValue',{ title: variable});
content string 否 該頁面的自定義內容值 
bat.track('CustomValue',{ content: variable});
time string 否 該頁面的自定義時間值
時間格式為24h		 
bat.track('CustomValue',{ time: variable});
category[i]
i為從1開始的正整數		 string 否 該頁面的自定義分類值 
bat.track('CustomValue',{ category1: variable,category2: variable});
label[i]
i為從1開始的正整數		 string 否 該頁面的自定義標籤值 
bat.track('CustomValue',{ label1: variable,label2: variable});
value string 否 該頁面的自定義值 
bat.track('CustomValue',{ value: variable});
goal string 否 該頁面的自定義目標值 
bat.track('CustomValue',{ goal: variable});
remark string 否 該頁面的自定義備註值 
bat.track('CustomValue',{ remark: variable});

使用範例

bat.track('CustomValue',{ title: 'LV 名牌包',content: '9999 元',time: '2017/01/01'});


bat.track('CustomValue',{ title: 'LV 名牌包',content: '9999 元',time: '2017/01/01 23:00',category1:'外套',catetory2:'大衣'});

按鈕點擊事件:buttonClick

名稱 值類型 是否必須提供 說明
category string 是 固定為 event
currentPage string 是 按鈕事件的當頁名稱
buttonName string 是 通常是用户與之互動的按鈕名稱（例如 'facebook'）
clickType string 否 互動類型（例如 '播放影片' / 暫停影片）
position string 否 按鈕的位置進行分類 固定為（'center / top / bottom / left / right'）
value string 否 與事件相關的值(自定義值)

buttonName一般規範：若您的按鈕屬於下列表格內類別請使用表格內的的buttonName以利資料分析

類別 buttonName 名稱

使用範例以Javascript為例

var linkelement = document.querySelector('element');//取得點擊元素
linkelement.addEventListener('click', function(e) { //建立監聽事件
   var linkname = e.target.innerText; // 點擊目標元素文字
   bat.track('buttonClick',{category:'event',currentPage:'購買商品頁面',buttonName: linkname});
});

使用範例以Jquery為例

$('.element').on('click',function(e){ // 點擊事件
   var linkname = $(this).text(); // 定義目標元素名稱
   bat.track('buttonClick',{category:'event',currentPage:'購買商品頁面',buttonName: linkname});
});
